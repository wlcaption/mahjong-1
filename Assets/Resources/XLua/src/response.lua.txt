local assert = assert
local errorcode = require "errorcode"
local cls = class("response")

function cls:ctor(env, clientsock, ... )
	-- body
	self._env = env
	self._clientsock = clientsock
	return self
end

function cls:handshake(args, ... )
	-- body
	assert(args.errorcode == errorcode.SUCCESS)
	printInfo("handshake")
end

return cls
