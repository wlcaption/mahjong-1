local assert = assert
local errorcode = require "errorcode"
local cls = class("response")

function cls:ctor(env, clientsock, ... )
	-- body
	self._env = env
	self._clientsock = clientsock
	return self
end

function cls:handshake(args, ... )
	-- body
	assert(args.errorcode == errorcode.SUCCESS)
	printInfo("handshake")
end

function cls:enter_room(args, ... )
	-- body
	local controller = self._env:get_controller("game")
	controller:enter_room(args)
end

function cls:leave_room(args, ... )
	-- body
	local controller = self._env:top()
	controller:leave_room(args)
end
